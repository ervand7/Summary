Нужно всегда обрабатывать ошибки. В Go нет конструкций try — catch,
поэтому для надёжности программы необходимы проверки if err != nil.
Текст ошибки должен содержать достаточно информации, чтобы понять, в
чём причина и как это исправить.

/* плохо */
if err != nil {
    return err
}

/* хорошо */
if err != nil {
    return fmt.Errorf("БД не инициализирована: %w", err)
}
